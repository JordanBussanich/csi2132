CREATE TABLE Member (
	MemberNumber Integer UNIQUE NOT NULL,
	Lastname VARCHAR,
	Firstname VARCHAR,
	email VARCHAR,
	password VARCHAR,
	PRIMARY KEY(MemberNumber));
	

CREATE TABLE BilingInformation (
	MemberNumber Integer NOT NULL,
	CardNumber Integer,
	Type VARCHAR,
	PRIMARY KEY(MemberNumber),
	FOREIGN KEY(CardNumber) REFERENCES Card
	On delete restrict
	on update cascade);

CREATE TABLE Card(
	CardNumber Integer NOT NULL,
	ExpDate Date,
	Name VARCHAR,
	PRIMARY KEY(CardNumber));

CREATE TABLE BilingAddress(
	BAddresID Integer NOT NULL,
	MemberNumber Integer,
	address1 VARCHAR,
	address2 VARCHAR,
	city VARCHAR,
	phone VARCHAR,
	postalcode VARCHAR,
	PRIMARY KEY(BAddresID),
	FOREIGN KEY (MemberNumber) REFERENCES Member
	On update cascade
	on delete restrict);

CREATE TABLE ShippingAddress(
	SAddresID Integer NOT NULL,
	MemberNumber Integer,
	address1 VARCHAR,
	address2 VARCHAR,
	city VARCHAR,
	phone VARCHAR,
	postalcode VARCHAR,
	PRIMARY KEY(SAddresID),
	FOREIGN KEY (MemberNumber) REFERENCES Member
	On update cascade
	On delete restrict);

CREATE TABLE Purchase(
	InvoiceNumber Integer NOT NULL,
	MemberNumber Integer,
	VideoID Integer,
	VideoID2 Integer,
	date_ordered DATE,
	date_shipped DATE,
	shipping_cost FLOAT,
	speed VARCHAR,
	carrier VARCHAR,
	Primary key (InvoiceNumber),
	FOREIGN KEY (MemberNumber) REFERENCES Member
	On delete restrict
	On update cascade,
	FOREIGN KEY (VideoID) REFERENCES Video
	On delete restrict
	On update cascade);

CREATE TABLE Returns(
	InvoiceNumber Integer,
	MemberNumber Integer,
	VideoID Integer,
	Negative_sales Numeric(8,2),
	Primary key(InvoiceNumber, VideoID),
	Foreign key (InvoiceNumber) references Purchase
	On delete restrict
	On update cascade,
	Foreign key(VideoID) references video
	On delete restrict
	On update cascade,
	Foreign key(MemberNumber) references Member
	On delete restrict
	On update cascade);
	
CREATE TABLE Download(
	MemberNumber Integer,
	VideoID Integer,
	Date DATE,
	Time TIME,
	Fee FLOAT,
	PRIMARY KEY(MemberNumber, VideoID),
	FOREIGN KEY(VideoID) REFERENCES Video
	On delete restrict
	On update cascade,
	FOREIGN KEY(MemberNumber) REFERENCES Member
	On delete restrict
	On update cascade);

CREATE TABLE Video(
	VideoID INTEGER UNIQUE NOT NULL,
	Name VARCHAR,
	Salesprice NUMERIC (8,2),
	Genre Varchar,
	Rating VARCHAR,
	InStock Boolean,
	DirectorID Integer,
	Year VARCHAR,
	Duration Integer,
	ActorID Integer UNIQUE NOT NULL,
	ActorID2 Integer,
	ActorID3 Integer,
	PRIMARY KEY (VideoID),
	FOREIGN KEY(DirectorID) REFERENCES Director
	On update cascade
	On delete restrict,
	Foreign key(ActorID) REFERENCES Actor
	On Update cascade
	On delete restrict);

CREATE TABLE Director(
	DirectorID Integer NOT NULL,
	Lastname VARCHAR,
	Firstname VARCHAR,
	Date_of_birth DATE,
	PRIMARY KEY(DirectorID));

CREATE TABLE Award(
	AwardID Integer NOT NULL,
	Year VARCHAR,
	Description VARCHAR,
	Category VARCHAR,
	PRIMARY KEY(AwardID, Year));

CREATE TABLE VideoAwards(
	VideoID Integer,
	AwardID Integer,
	Year VARCHAR,
	Won Boolean,
	PRIMARY KEY(VideoID, AwardID, Year),
	FOREIGN KEY (VideoID) REFERENCES Video
	On delete restrict
	On update cascade,
	FOREIGN KEY(AwardID, YEAR) REFERENCES Award
	On delete restrict
	On update cascade);

CREATE TABLE Actor(
	ActorID Integer,
	Lastname VARCHAR,
	Firstname VARCHAR,
	Date_of_birth DATE,
	IMdB_link TEXT,
	PRIMARY KEY (ActorID));

CREATE TABLE VideoStar(
	VideoID Integer,
	ActorID Integer,
	Rolename VARCHAR,
	FOREIGN KEY(VideoID) REFERENCES Video
	On delete restrict
	On update cascade,
	FOREIGN KEY(ActorID) REFERENCES Actor
	On delete restrict
	On update cascade);
